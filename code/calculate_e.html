#include <stdio.h>
#include <stdlib.h>

double calculate_e(int n){

    double temp_e;

    int i = n;

    double factorialIngevoerdeWaarde = 1;

     while(n>0){            //bereken factorial van de ingevoerde waarde

        factorialIngevoerdeWaarde *= n;     // FIW = 16 * n * n-1 * n-2 ... n-?=0
        n-- ;
    }

    temp_e = 1/factorialIngevoerdeWaarde-1;
    double a = factorialIngevoerdeWaarde;

    while(i>0){


          a = a/i;
          temp_e += 1/a;


        i--;
    }


    return(1+temp_e);
}
